AUTHORS

CREATE TABLE  "AUTHORS" 
   (    "AUTHOR_ID" NUMBER(7,0), 
        "AFIRST_NAME" VARCHAR2(255) NOT NULL ENABLE, 
        "ALAST_NAME" VARCHAR2(255) NOT NULL ENABLE, 
        "A_EMAIL" VARCHAR2(255) NOT NULL ENABLE, 
         CONSTRAINT "AUTHORS_PK" PRIMARY KEY ("AUTHOR_ID")
  USING INDEX  ENABLE
   )

CREATE TABLE  "BOOKS" 
   (    "BOOK_ID" NUMBER(7,0) NOT NULL ENABLE,

         CONSTRAINT "BOOKS_PK" PRIMARY KEY ("BOOK_ID")

/
ALTER TABLE  "BOOKS" ADD CONSTRAINT "BOOKS_FK" FOREIGN KEY ("AUTHOR_ID")
